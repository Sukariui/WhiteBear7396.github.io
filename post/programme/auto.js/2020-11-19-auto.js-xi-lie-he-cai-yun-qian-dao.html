    【Auto.js系列】和彩云签到

    为什么选择Auto.js

    Auto.js 是个基于 JavaScript 语言运行在Android平台上的脚本框架。Auto.js主要工作原理是基于辅助服务AccessibilityService。Auto.js是利用安卓系统的“辅助功能”实现类似于按键精灵一样，可以通过代码模拟一系列界面动作的辅助工作。因为是开源框架所以安全性很高，他能在手机上模拟人的重复繁琐的工作，不打破被执行的APP规则，不修改，不破坏被执行的APP，可以放心使用。官方的入门文档写得比较全面，可以进行直接上手。

function permissions() {
    if (auto.service == null) {
        toast("请开启本软件的无障碍服务");
        auto.waitFor();
        return
    }
    sleep(200);
    if (!floaty.checkPermission()) {
        toast("请开启悬浮窗和后台弹出界面权限");
        floaty.requestPermission();
        return
    }
}//获取权限函数

function delay(seconds){
    sleep(1000*seconds);
}//延时函数

function RunApp() 
{   
    toastLog("本设备的分辨率为 \n宽："+device.width+"\n长："+device.height);//输出设备的分辨率
    setScreenMetrics(1440, 3040); //设定以下坐标点击的基准屏幕分辨率
    var apprun = currentActivity();
    if (apprun != "com.chinamobile.mcloud") {
        launch("com.chinamobile.mcloud");
        console.log("和彩云启动中...");
        text("首页").waitFor();
        toastLog("到达首页");
        console.log("到达首页");

    if (!homepage) {
        toastLog("没有找到首页，重启！");
        back();
        return RunApp();

}//启动App函数

                    window.text.setText(tstr);

                    window.message.setText(lastLog);             
 
                })
        },1000)
    });//显示运行时间
}//启动悬浮窗

function brighten_the_screen(){
    if(!device.isScreenOn()){
      device.wakeUpIfNeeded();//如果手机锁屏，就唤醒屏幕
    }else {
        console.log("手机已解锁，开始运行。")
        return;//跳出当前函数，进入下一个函数
    }
    sleep(2000);
    unlock();
    sleep(300);
    for(var i=0;i<password.length;i++){
      a=password.charAt(i)
      //log(a)
      sleep(500);
      b=text(a).findOne().bounds()
      click(b.centerX(),b.centerY())
    }
} //亮屏函数
function unlock(){
    var xyArr = [220];
    var x0=device.width/2;
    var y0=device.height/4*3;
    var angle = 0;
    var x = 0;
    var y = 0;
    for (let i = 0; i < 30; i++) {
      y = x * tan(angle)
      //log(y)
      if((y0-y)<0){
        break
      }
      var xy = [x0+x,y0-y]
      xyArr.push(xy)
      x += 5;
      angle += 3
    }
    gesture.apply(null,xyArr)
    function tan(angle) {
      return Math.tan(angle * Math.PI / 180);
    }
}//解锁函数
